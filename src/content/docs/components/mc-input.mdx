## Usage

```tsx
import { McInput } from "@/components/ui/mc-input"
import { Mail } from "lucide-react"

export default function Example() {
  return (
    <McInput 
      placeholder="Enter your email" 
      leftIcon={<Mail className="size-4" />}
    />
  )
}
```

## Examples

### Variants

```tsx
<McInput variant="default" placeholder="Default variant" />
<McInput variant="filled" placeholder="Filled variant" />
<McInput variant="outline" placeholder="Outline variant" />
```

### With Icons

```tsx
import { Search, Mail } from "lucide-react"

<McInput 
  leftIcon={<Search className="size-4" />} 
  placeholder="Search components..." 
/>

<McInput 
  rightIcon={<Mail className="size-4" />} 
  placeholder="Contact support" 
/>
```

### Validation States

The component supports standard HTML validation attributes and ARIA states.

```tsx
<McInput aria-invalid="true" placeholder="Invalid input" />
<McInput required placeholder="Required field" />
<McInput disabled placeholder="Disabled field" />
```

## API Reference

### Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `"default" \| "filled" \| "outline"` | `"default"` | The visual style of the input |
| `leftIcon` | `ReactNode` | `undefined` | Icon element to render inside the input on the left |
| `rightIcon` | `ReactNode` | `undefined` | Icon element to render inside the input on the right |
| `className` | `string` | `undefined` | Additional CSS classes for the input wrapper |

All other props from `@base-ui/react/input` are also supported.
